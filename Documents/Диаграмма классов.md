classDiagram
    class User {
        +id: UUID
        +email: String
        +passwordHash: String
        +role: Enum (Customer, Seller, Admin)
        +createdAt: DateTime
        +updatedAt: DateTime
    }

    class Customer {
        +userId: UUID (FK → User)
        +name: String
        +phone: String
        +addresses: Address[]
        +avatarUrl: String?
    }

    class Seller {
        +userId: UUID (FK → User)
        +companyName: String
        +description: String
        +contactEmail: String
        +contactPhone: String
        +logoUrl: String?
    }

    class Product {
        +id: UUID
        +sellerId: UUID (FK → Seller)
        +name: String
        +description: String
        +category: String
        +brand: String
        +createdAt: DateTime
        +updatedAt: DateTime
    }

    class Variant {
        +id: UUID
        +productId: UUID (FK → Product)
        +type: String
        +value: String
    }

    class Tag {
        +id: String
    }

    class ProductTag {
        +productId: UUID (FK → Product)
        +tagId: Sting(FK → Tag)
    }

    class Review {
        +id: UUID
        +productId: UUID (FK → Product)
        +customerId: UUID (FK → Customer)
        +rating: Integer (1-5)
        +comment: String?
        +photos: String[]?
        +createdAt: DateTime
    }

    class Favourite {
        +customerId: UUID (FK → Customer)
        +productId: UUID (FK → Product)
    }

    class Cart {
        +id: UUID
        +customerId: UUID (FK → Customer)
    }

    class CartItem {
        +cartId: UUID (FK → Cart)
        +variantId: UUID (FK → Variant)
        +quantity: Integer
    }

    class Order {
        +id: UUID
        +customerId: UUID (FK → Customer)
        +status: Enum (Pending, Paid, Shipped, Delivered, Cancelled)
        +total: Decimal
        +createdAt: DateTime
    }

    class OrderItem {
        +orderId: UUID (FK → Order)
        +variantId: UUID (FK → Variant)
        +quantity: Integer
        +price: Decimal
    }


    User <|-- Customer
    User <|-- Seller
    Seller "1" --> "*" Product
    Product "1" --> "*" Variant
    Product "*" --> "*" Tag : через ProductTag
    Product "1" --> "*" Review
    Customer "1" --> "*" Review
    Customer "1" --> "*" Favourite
    Customer "1" --> "*" Cart
    Cart "1" --> "*" CartItem
    Customer "1" --> "*" Order
    Order "1" --> "*" OrderItem
    Variant "1" --> "*" OrderItem
    Variant "1" --> "*" CartItem
